\section{Article 9 - Directors}
\label{article-9---directors}

\subsection{Purpose}
\label{purpose-directors}

\begin{enumerate}
 \item The Directors shall be the directors formally recognized by the Government of Canada and financial institutions.
\end{enumerate}

\subsection{Composition}
\label{composition-directors}

\begin{enumerate}
 \item The Directors of the Federation are composed of eight (8) members.
 \item The Directors are as follows:
    \begin{enumerate}
    \item The President;
    \item The Vice President Finance and Administration;
    \item The four (4) Regional Ambassadors;
    \item The two (2) National Councillors;
    \end{enumerate}
\end{enumerate}
