\section{Article 7 - Regional Ambassadors}
\label{article-7---regional-ambassadors}

\subsection{Composition}
\label{composition-regional-ambassadors}

\begin{enumerate}
 \item The Regional Ambassadors are chosen by their respective region to represent on the Board.
 \item There is one Regional Ambassador per region, and they have tasks to accomplish at both the national and regional levels.
\end{enumerate}

\subsection{Term}
\label{term-regional-ambassadors}

\begin{enumerate}
 \item The term of office for a Regional Ambassador shall be determined to be the same as the CFES fiscal year, unless the relevant regional organization deems otherwise.
\end{enumerate}
