\section{Article 4 - General Meetings}
\label{article-4---general-meetings}

\subsection{Purpose}
\label{purpose-meetings}

\begin{enumerate}
 \item The regularly scheduled Annual General Meeting and additional General Meetings serve the purpose of allowing Members to examine the activities and the accountability of the Federation.
 \item Extraordinary General Meetings will be convened by resolution of the Board or by request of the Members, as outlined in Section 7.6.1 of the Bylaws.
\end{enumerate}

\subsection{Annual Meetings}
\label{annual-meetings}

\begin{enumerate}
 \item The Annual General Meeting shall be held at the Canadian Engineering Leadership Conference.
 \item A General Meeting shall be held at the Canadian Engineering Spring Summit.
 \item A General Meeting shall be held at the Summit on Development of Engineering Societies.
\end{enumerate}
