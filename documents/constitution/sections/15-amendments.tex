\section{Article 15 - Amendments}
\label{article-15---amendments}

\subsection{Substantive Amendments}
\label{substantive-amendments}

\begin{enumerate}
 \item The Constitution of the Federation may be repealed or amended by resolution at a General Meeting where no less than two thirds (2/3) active members present must agree.
\end{enumerate}

\subsection{Grammatical Amendments}
\label{grammatical-amendments}

\begin{enumerate}
 \item Amendments to this constitution that are simply corrective measures to ensure accurate translation and grammatical structure may be passed by the Board with a two thirds (2/3) majority, provided these amendments do not change the concept or spirit of the point being amended.
\end{enumerate}

\subsection{Temporary Amendments}
\label{temporary-amendments}

\begin{enumerate}
 \item The Board may prescribe rules and regulations consistent with this Constitution relating to the management and operation of the Federation as they deem expedient.
 \item The rules and regulations shall have force and effect only until the next General Meeting when they shall be confirmed.
 \item If the rules and regulation fail to be confirmed, they shall at and from that time cease to have any force and effect.
 \item Any amendments passed by the Board in this way shall be communicated to all Members within 2 business days.
\end{enumerate}
