\section{Article 2 - Definitions}
\label{article-2---definitions}

\begin{tblr}{colspec={lX}, rowsep=4pt}
  \textbf{Board Members}       & Members of the Board, as specified in Article 10. \\
  \textbf{Members}             & Member societies, as specified in Article 4. \\
  \textbf{National Executive}  & President and Vice-Presidents as specified in Article 6. \\
  \textbf{Resolution}          & Motion presented as ``Proposal'' and approved by a majority vote. \\
  \textbf{Special Resolution}  & Motion presented as ``Special Proposal'' and approved by a supermajority vote. \\
  \textbf{Signing Officer}     & Person who has the right to authorize financial transactions on behalf of the CFES, as specified in Article 13. \\
  \textbf{General Assembly}    & Meeting of the Members. \\
\end{tblr}
