\section{Article 14 - Dissolution of the Federation}
\label{article-14---dissolution-of-the-federation}

\begin{enumerate}
 \item The Federation will be dissolved upon resolution at a meeting of the Membership where no less than two thirds (2/3) of the Member Societies must agree.
 \item All liabilities will be honoured and the remaining assets will be distributed to one or more organizations with goals similar to the Federation.
\end{enumerate}
