\section{Article 11 - Elections}
\label{article-11---elections}

\subsection{General Policy}
\label{general-policy}

\begin{enumerate}
 \item The election procedures outlined in the CFES Policy Manual are valid for the election in the General Assembly or by the Board for the National Councillors, the National Executive and the Regional Ambassadors.
 \item The election procedures must include a stipulation that, to be elected, a candidate must receive the absolute majority of the vote, abstentions excluded.
\end{enumerate}

\subsection{Eligibility}
\label{eligibility}

\begin{enumerate}
 \item To be eligible, the candidate for a position on the National Executive must be a member of an active member at the time of its candidature.
 \item There is no specific eligibility condition for the candidate for a National Councillor position.
 \item No student shall hold more than one position within the CFES Team at any given time.
\end{enumerate}
