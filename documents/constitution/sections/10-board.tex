\section{Article 10 - Board}
\label{article-10---board}

\subsection{Purpose}
\label{purpose-board}

\begin{enumerate}
 \item A Board shall manage the property and business of the CFES.
\end{enumerate}

\subsection{Composition}
\label{composition-board}

\begin{enumerate}
 \item The Board of the Federation shall be composed of seven (7) members, hereafter referred to as Voting Board Members:
\end{enumerate}

\begin{enumerate}
 \item[a.] Regional Ambassadors (4)
 \item[b.] National Councillors (2)
 \item[c.] President (1)
  \begin{enumerate}
   \item The Chair, or the Vice-Chair if fulfilling the duties of the Chair, shall only vote when there is a tie.
   \item The Board shall be composed of the following non-voting members, hereafter referred to as Non-Voting Board Members:
  \end{enumerate}
\end{enumerate}

\begin{enumerate}
 \item[a.] The Board Chair;
 \item[b.] The Vice President Academic;
 \item[c.] The Vice President Finance and Administration;
 \item[d.] The Vice President External;
 \item[e.] The Vice President Services;
 \item[f.] The Vice President Communications;
 \item[g.] Any ratified incoming representative for the Board Membership.
\end{enumerate}
