\section{Article 10 - Board}
\label{article-10---board}

\subsection{Purpose}
\label{purpose-board}

\begin{enumerate}
 \item A Board shall manage the property and business of the CFES.
\end{enumerate}

\subsection{Composition}
\label{composition-board}

\begin{enumerate}
 \item The Board of the Federation shall be composed of seven (7) members, hereafter referred to as Voting Board Members:
  \begin{enumerate}
   \item Regional Ambassadors (4)
   \item National Councillors (2)
   \item President (1)
  \end{enumerate}
 \item The Chair, or the Vice-Chair if fulfilling the duties of the Chair, shall only vote when there is a tie.
 \item The Board shall be composed of the following non-voting members, hereafter referred to as Non-Voting Board Members:
  \begin{enumerate}
   \item The Board Chair;
   \item The Vice President Academic;
   \item The Vice President Finance and Administration;
   \item The Vice President External;
   \item The Vice President Services;
   \item The Vice President Communications;
   \item Any ratified incoming representative for the Board Membership.
  \end{enumerate}
\end{enumerate}
