\section{Article 10 - Team}
\label{article-10---team}

\subsection{Purpose}
\label{purpose-team}

\begin{enumerate}
 \item The CFES Team shall participate in the workings of the Federation.
\end{enumerate}

\subsection{Composition}
\label{composition-team}

\begin{enumerate}
 \item The CFES Team shall be composed of:
  \begin{enumerate}
   \item The Board;
   \item Activity Managers, as defined in Article 12.2.2;
   \item Commissioners, as defined in the Bylaws;
  \end{enumerate}
\end{enumerate}
