\section{Article 6 - National Executive}
\label{article-6---national-executive}

\subsection{Composition}
\label{composition-national-executive}

\begin{enumerate}
 \item The National Executive shall consist of the President, the Vice President Academic, the Vice President Finance and Administration, the Vice President External, and the Vice President Services.
 \item Powers and duties of members of the National Executive shall be found in the Policy Manual of the Federation.
\end{enumerate}

\subsection{Election}
\label{election-national-executive}

\begin{enumerate}
 \item The National Executive is elected at the Annual General Assembly.
\end{enumerate}

\subsection{Term of Office}
\label{term-of-office}

\begin{enumerate}
 \item The term of office of the National Executive is the same as the fiscal year of the Federation.
 \item Concluding their term in office, the Vice-President Finance and Administration will transition to the role of Outgoing Vice-President Finance and Administration, with a term concluding at the end of August.
\end{enumerate}
