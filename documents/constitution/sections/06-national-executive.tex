\section{Article 5 - National Executive}
\label{article-5---national-executive}

\subsection{Composition}
\label{composition-national-executive}

\begin{enumerate}
 \item The National Executive shall consist of the:
  \begin{enumerate}
   \item President
   \item Vice President Academic
   \item Vice President Communications
   \item Vice President External
   \item Vice President Finance and Administration
   \item Vice President Services
  \end{enumerate}
 \item Powers and duties of the National Executive shall be found in the Policy Manual.
\end{enumerate}

\subsection{Election}
\label{election-national-executive}

\begin{enumerate}
 \item The National Executive is elected at the Annual General Meeting.
\end{enumerate}

\subsection{Term of Office}
\label{term-of-office}

\begin{enumerate}
 \item The term of office of the National Executive is the same as the Fiscal Year.
 \item Concluding their term in office, the Vice-President Finance and Administration will transition to the role of Outgoing Vice-President Finance and Administration, with a term concluding at the end of August.
\end{enumerate}
