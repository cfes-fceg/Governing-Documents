\section{Article 1 - General}
\label{article-1---general}

\subsection{Name of the Corporation}
\label{name-of-the-corporation}

\begin{enumerate}
 \item The name of the Corporation shall be ``The Canadian Federation of Engineering Students'', hereafter called ``the Federation''.
 \item The official French name of the Federation shall be “La Fédération canadienne étudiante en génie”.
\end{enumerate}

\subsection{Head Office}
\label{head-office}

\begin{enumerate}
 \item The Head Office of the Federation shall be located at the office of Engineers Canada, 55 Metcalfe Street -- Suite 300, Ottawa, Ontario, K1P 6L5.
\end{enumerate}

\subsection{The Seal}
\label{the-seal}

\begin{enumerate}
 \item The Seal, an impression whereof is stamped below, shall be the Seal of the Federation.
\end{enumerate}

\subsection{Fiscal Year}
\label{fiscal-year}

\begin{enumerate}
 \item The Fiscal Year of the Federation shall conclude on April 30 each year.
\end{enumerate}

\subsection{Abbreviation}
\label{abbreviation}

\begin{enumerate}
 \item “CFES” and “FCEG” are employed to designate the Canadian Federation of Engineering Students.
\end{enumerate}

\subsection{Logo}
\label{logo}

\begin{enumerate}
 \item The Logo that appears below is employed to designate the Canadian Federation of Engineering Students.
\end{enumerate}
