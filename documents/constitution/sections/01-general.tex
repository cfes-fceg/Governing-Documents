\section{Article 1 - General}
\label{article-1---general}

\subsection{Name of the Corporation}
\label{name-of-the-corporation}

\begin{enumerate}
 \item The name of the Corporation shall be ``The Canadian Federation of Engineering Students'', hereafter called ``the CFES''.
\end{enumerate}

\subsection{Head Office}
\label{head-office}

\begin{enumerate}
 \item The Head Office of the CFES shall be located at the office of Engineers Canada, 55 Metcalfe Street -- Suite 300, Ottawa, Ontario, K1P 6L9.
\end{enumerate}

\subsection{The Seal}
\label{the-seal}

\begin{enumerate}
 \item The Seal, an impression whereof is stamped below, shall be the Seal of the Federation.
\end{enumerate}

\subsection{Fiscal Year}
\label{fiscal-year}

\begin{enumerate}
 \item The fiscal year of the CFES shall conclude on 31 March each year.
\end{enumerate}

\subsection{Abbreviations and Logo}
\label{abbreviations-and-logo}

\subsubsection{Canadian Federation of Engineering Students}
\label{canadian-federation-of-engineering-students-cfes}

\begin{enumerate}
 \item The abbreviations FCEG and CFES and the logo that appears below are employed to designate the Canadian Federation of Engineering Students.
 \item In this constitution, ``Federation'' designates the Canadian Federation of Engineering Students.
\end{enumerate}

\subsubsection{Canadian Engineering Competition}
\label{canadian-engineering-competition}

\begin{enumerate}
 \item The abbreviations CCI and CEC are employed to designate the Canadian Engineering Competition.
 \item The Canadian Engineering Competition logo must incorporate, in part or in full, the CFES logo as established in Article 1.5.1.
\end{enumerate}

\subsubsection{Canadian Engineering Leadership Conference}
\label{canadian-engineering-leadership-conference}

\begin{enumerate}
 \item The abbreviation CELC is employed to designate the Canadian Federation of Engineering Students Canadian Engineering Leadership Conference.
 \item The Canadian Federation of Engineering Students Canadian Engineering Leadership Conference logo must incorporate, in part or in full, the CFES logo as established in Article 1.5.1.
\end{enumerate}

\subsubsection{Conference on Diversity in Engineering}
\label{conference-on-diversity-in-engineering}

 \item The Canadian Federation of Engineering Students Conference on Diversity in Engineering logo must incorporate, in part or in full, the CFES logo as established in Article 1.5.1.
\end{enumerate}
