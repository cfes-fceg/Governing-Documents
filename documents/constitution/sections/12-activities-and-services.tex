\section{Article 12 - Activities and Services}
\label{article-12---activities-and-services}

\subsection{Overview}
\label{overview}

\subsubsection{Canadian Engineering Competition}
\label{canadian-engineering-competition}
\begin{enumerate}

 \item The abbreviation CEC is employed to designate the Canadian Engineering Competition.
 \item The French name of CEC shall be “Compétition canadienne d’ingénierie”, abbreviated as CCI.
\end{enumerate}

\subsubsection{Canadian Engineering Leadership Conference}
\label{canadian-engineering-leadership-conference}
\begin{enumerate}

 \item The abbreviation CELC is employed to designate the Canadian Engineering Leadership Conference.
 \item The French name of CELC shall be “Congrès canadien sur le leadership en ingénierie”, abbreviated as CCLI.
\end{enumerate}

\subsubsection{Conference on Diversity in Engineering}
\label{conference-on-diversity-in-engineering}
\begin{enumerate}

 \item The abbreviation CDE is employed to designate the Conference on Diversity in Engineering.
 \item The French name of CDE shall be “Congrès sur la diversité en ingénierie”, abbreviated as CDI.
\end{enumerate}

\subsubsection{Summit on Development of Engineering Societies}
\label{summit-on-development-of-engineering-societies}
\begin{enumerate}

 \item The abbreviation SDES is employed to designate the Summit on Development of Engineering Societies.
 \item The French name of SDES shall be “Sommet du développement des associations en ingénierie”, abbreviated as SDAI.
\end{enumerate}

\subsubsection{Conference on Sustainability in Engineering}
\label{conference-on-sustainability-in-engineering}
\begin{enumerate}

 \item The abbreviation CSE is employed to designate the Conference on Sustainability in Engineering.
 \item The French name of CSE shall be “Congrès sur le développement durable en ingénierie”, abbreviated as CDDI.
\end{enumerate}

\subsection{Organization of Activities and Services}
\label{organization-of-activities-and-services}

\subsubsection{Logos}
\label{logos}
\begin{enumerate}
 \item The logo of Activities and Services must incorporate, in part or in full, the logo of the Federation as established in Article 1.5.1.
\end{enumerate}

\subsubsection{Activity Agreements}
\label{activity-agreements}

\begin{enumerate}

 \item The specifications of responsibilities between the Federation and the Activities are outlined in their respective CFES Activity Agreements.
\end{enumerate}

\subsubsection{Organizing Members}
\label{organizing-members}

\begin{enumerate}
 \item Only Member Societies, as outlined in Article 3.1, and the Board, can organize the activities and services of the Federation.
 \item Organizing Members are appointed at any General Meeting.
\end{enumerate}

\subsubsection{Activity Managers}
\label{activity-managers}

\begin{enumerate}
 \item The Activity Managers are appointed by the Organizing Member.
 \item Their term of office spans from the time of appointment until the resolution of any accounts receivable or payable at the conclusion of the activity in question.
 \item They have the exclusive responsibility of all that is related to their activity, including, but without limitation to, the financial accountability and their audition by an external auditor, the appointment of an organising committee, the signature of official documents, etc.
 \item They answer directly to their Advisory Board, the Organizing Member, and to the Board.
\end{enumerate}
