\section{Article 2 - Les définitions}\label{article-2---les-duxe9finitions}

\begin{tblr}{colspec={lX}, rowsep=4pt}
  \textbf{Le Conseil d\textquotesingle administration} & Membres du Conseil d\textquotesingle administration, tel que spécifié au Article 10. \\
  \textbf{Les membres}              & Les sociétés membres, telles que spécifiées au Article 4. \\
  \textbf{L'exécutif national}      & Le/la président.e et les vice-président.e.s, comme spécifié au Article 6. \\
  \textbf{Une résolution}           & Motion présentée sous le nom de ``Proposition'' et approuvée par un vote à la majorité. \\
  \textbf{Une résolution spéciale}  & Motion présentée sous le nom de ``Proposition spéciale'' et approuvée par un vote à une supermajorité. \\
  \textbf{Un agent de signature}    & Personne ayant le droit d\textquotesingle autoriser les transactions financières au nom de la FCÉG, comme spécifié au Article 13. \\
  \textbf{L'Assemblée générale}     & Réunion des membres. \\
\end{tblr}
